ğŸ“Š TimescaleDB Schema

Signal Chunk Entity

@Entity('signal_chunks')
@Index(['patientId', 'channel', 'tStart'])
export class SignalChunk {
  @PrimaryGeneratedColumn('uuid')
  id!: string;
  
  @Column()
  patientId!: string;
  
  @Column()
  channel!: string; // "EEG:C3", "ECG:leadI"
  
  @Column({ type: 'timestamptz' })
  tStart!: Date;
  
  @Column({ type: 'timestamptz' })
  tEnd!: Date;
  
  @Column({ type: 'real' })
  fs!: number; // Sampling frequency
  
  @Column('double precision', { array: true })
  values!: number[]; // Signal data
}

Quick Start

Get up and running in minutes

Using Docker (Recommended)

1. Clone and navigate to the project:

cd biotensor-lab

2. Copy environment files:

cp apps/web/.env.example apps/web/.env
cp apps/api/.env.example apps/api/.env
cp apps/inference/.env.example apps/inference/.env

3. Start all services:

docker compose up --build

4. Access the platform:

Web App
localhost:3000 API Health
localhost:4000 Inference Docs
localhost:8000
