Developer Experience

Python SDK Example synapse_sdk.py 
# OmniFlow Studio Python SDK
from synapse_sdk import Workflow, DataSource, Process, Sink

# Create workflow
wf = Workflow("oscilloscope_analysis")

# Add nodes
osc = DataSource.Oscilloscope(
    channels=4,
    sample_rate="1MS/s"
)
filter = Process.LowPassFilter(cutoff=20)
writer = Sink.HDF5Writer("data.h5")

# Connect nodes
wf.connect(osc >> filter >> writer)

# Execute
result = wf.run()
Rust HAL Driver keysight_driver.rs 
// Rust HAL driver implementation
use synapse_hal::{Device, Result};

#[derive(Debug)]
pub struct KeysightOscilloscope {
    session: ScpiSession,
}

impl Device for KeysightOscilloscope {
    async fn read_waveform(&mut self, ch: u8) -> Result> {
        let data = self.session
            .query(format!(":WAV:DATA? CHAN{}", ch))
            .await?;
        Ok(parse_binary(data))
    }
}
